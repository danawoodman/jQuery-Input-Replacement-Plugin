jQuery Input Replacement Plugin Demos

Example 1: No options set

$('#example1').input_replacement();

Example 2: prefocus_class set

$('#example2').input_replacement({
    prefocus_class: "prefocus"
});

Example 3: prefocus_class and default text set

$('#example3').input_replacement({
    text: "Default text...",
    prefocus_class: "prefocus"
});

Example 4: Field has value="Foobar" set

$('#example4').input_replacement({
    text: "Default text...",
    prefocus_class: "prefocus"
});
